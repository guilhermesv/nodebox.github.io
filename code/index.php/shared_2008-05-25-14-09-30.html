Posted by Lucas on May 27, 2008

Hello Gabra,

try this:

size(500,500)
ximport("cornu")
translate(0,0)
font("American Typewriter", 100)
path = textpath("2008", 0, 0)
 
ptlist = []
 
d = 6.15
for i in range(3):
    for pt in path:
        if pt.cmd == LINETO or pt.cmd == CURVETO:
            ptlist.append((pt.x+random(-d, d), pt.y+random(-d, d)))
 
nofill()
stroke(0)
cp = cornu.path(ptlist)
cp.fit(0, 0, 500, 500)
 
cp.draw()
good luck
